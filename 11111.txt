In the next.js program, I have created the following file, which is responsible
for fetching data:

// api/getPosts.js
import useSWR from 'swr';

const fetcher = (url) => fetch(url).then((res) => res.json());

export function getPosts() {
    const apiUrl = 'http://localhost/test.raiatec.com/wp-json/wp/v2/posts?per_page=20';
    const { data, error } = useSWR(apiUrl, fetcher);

    return {
        posts: data,
        isLoading: !error && !data,
        isError: error,
    };
}


To manage dynamic routing for individual posts based on their slug, please
Create the pages/[post-slug].js file and add the necessary codes to the api/getPosts.js file.
